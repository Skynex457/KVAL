# Обновление системы - установка всех последних обновлений безопасности
sudo apt update && sudo apt upgrade -y

# Установка принтера: sudo apt install cups-pdf

# Настройка SSH - сетевой протокол для удаленного управления
# Обоснование: позволяет подключаться к компьютеру удаленно для администрирования
sudo apt install -y openssh-server
sudo systemctl enable ssh
sudo systemctl start ssh
sudo systemctl status ssh

# Удаленное подключение к активной сессии
hostname -I - узнаем ip
ssh -X имя_пользователя@ip-адресс

# Установка базового ПО
# LibreOffice устанавливается через центр приложений
sudo apt install vim git zip htop gnome-calculator

# Резервное копирование
sudo mkdir -p /backup
sudo tar -czvf /ubuntu_backup_$(date +%Y%m%d).tar.gz \
  --exclude=/ubuntu_backup_*.tar.gz \
  --exclude=/proc/ \
  --exclude=/sys/ \
  --exclude=/dev/ \
  --exclude=/mnt/ \
  --exclude=/media/ \
  --exclude=/tmp/ \
  --exclude=/run/ \
  --exclude=/lost+found/ \
  --exclude=/var/cache/apt/archives/*.deb \
  --exclude=/home/*/.cache \
  --exclude=/var/log/*.log \
  --one-file-system \
  /
ls -lh /backup/
