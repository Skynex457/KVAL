#!/bin/bash
# ================================================================
# ЭКЗАМЕНАЦИОННЫЙ БИЛЕТ №0 - ПОЛНОЕ ВЫПОЛНЕНИЕ
# ПМ.04 Сопровождение и обслуживание программного обеспечения

# ================================================================
# ЧАСТЬ 1: УСТАНОВКА ОС, НАСТРОЙКА И КОНФИГУРИРОВАНИЕ
# ================================================================

# 1.1. Настройка параметров ОС, драйверов и служб
# ----------------------------------------------------------------

# Обновление системы - установка всех последних обновлений безопасности
sudo apt update && sudo apt upgrade -y

# Утилиты для настройки ядра (sysfsutils - управление параметрами ядра, tuned - профили производительности)
# Обоснование: позволяют оптимизировать работу системы под задачи разработки
sudo apt install -y sysfsutils tuned
sudo tuned-adm profile latency-performance

# Настройка SSH - сетевой протокол для удаленного управления
# Обоснование: позволяет подключаться к компьютеру удаленно для администрирования
sudo apt install -y openssh-server
sudo systemctl enable ssh
sudo systemctl start ssh
sudo systemctl status ssh

# Настройка удаленного доступа к активной сессии (x11vnc)
# Обоснование: позволяет видеть текущий рабочий стол удаленно, для помощи и демонстрации
sudo apt install -y x11vnc
x11vnc -storepasswd         # Ввести пароль 123 и подтвердить
x11vnc -display :0 -auth guess -forever -usepw &

# 1.2. Настройка интернет-соединения
# ----------------------------------------------------------------

# Выбор правильного сетевого интерфейса - показываем какой интерфейс используется
ip a
ip link show
ip route | grep default

# Проверка сетевого соединения - демонстрация работы интернета
ping -c 4 8.8.8.8
ping -c 4 google.com

# Настройка статического IP (если нужно)
# sudo nano /etc/netplan/01-network-manager-all.yaml
# network:
#   ethernets:
#     enp0s3:
#       dhcp4: no
#       addresses: [192.168.1.100/24]
#       nameservers:
#         addresses: [8.8.8.8, 1.1.1.1]
#   version: 2
# sudo netplan apply

# 1.3. Установка базового ПО
# ----------------------------------------------------------------
# Обоснование: стандартный набор инструментов для работы техника-программиста
sudo apt install -y vim git curl wget htop neofetch zip unzip p7zip-full build-essential mc

# 1.4. Установка виртуального принтера
# ----------------------------------------------------------------
# Обоснование: позволяет печатать документы в PDF без реального принтера
sudo apt install -y cups cups-pdf
sudo systemctl enable cups
sudo systemctl start cups
sudo systemctl status cups
sudo lpadmin -p Virtual_PDF -E -v cups-pdf:/ -m everywhere
sudo lpadmin -d Virtual_PDF
lpstat -p -d
echo "Test print" | lp -d Virtual_PDF
ls ~/PDF/

# 1.5. Установка VirtualBox (эмулятор ОС на ПК)
# ----------------------------------------------------------------
# Обоснование: позволяет запускать другие ОС внутри основной для тестирования
wget -O- https://www.virtualbox.org/download/oracle_vbox_2016.asc | sudo gpg --dearmor --yes --output /usr/share/keyrings/oracle-virtualbox-2016.gpg
echo "deb [arch=amd64 signed-by=/usr/share/keyrings/oracle-virtualbox-2016.gpg] https://download.virtualbox.org/virtualbox/debian $(lsb_release -cs) contrib" | sudo tee /etc/apt/sources.list.d/virtualbox.list
sudo apt update
sudo apt install -y virtualbox-7.0
sudo usermod -aG vboxusers $USER
vboxmanage --version

# 1.6. Установка эмулятора 3D графики (Mesa + Vulkan)
# ----------------------------------------------------------------
# Обоснование: позволяет тестировать 3D-приложения без реального GPU
sudo apt install -y mesa-utils mesa-utils-extra vulkan-tools libvulkan1 mesa-vulkan-drivers
sudo usermod -aG video $USER
glxinfo | grep "OpenGL version"
vulkaninfo | grep "deviceName"
glxgears &

# ================================================================
# ЧАСТЬ 2: ПРИМЕНЕНИЕ СРЕДСТВ ЗАЩИТЫ КОМПЬЮТЕРНЫХ СИСТЕМ
# ================================================================

# 2.1. Резервное копирование ОС
# ----------------------------------------------------------------
# Обоснование: создание бэкапа для восстановления системы в случае сбоя
sudo mkdir -p /backup
sudo tar -czpvf /backup/ubuntu_backup_$(date +%Y%m%d).tar.gz \
  --exclude=/backup \
  --exclude=/proc \
  --exclude=/sys \
  --exclude=/dev \
  --exclude=/mnt \
  --exclude=/media \
  --exclude=/tmp \
  --exclude=/run \
  --exclude=/lost+found \
  --exclude=/var/cache/apt/archives/*.deb \
  /
ls -lh /backup/

# 2.2. Создание установочного образа системы
# ----------------------------------------------------------------
# Обоснование: образ для установки системы на другие компьютеры
sudo apt install -y genisoimage
mkdir -p ~/livecd
echo "Ubuntu System Image $(date)" > ~/livecd/readme.txt
sudo genisoimage -o /backup/system_image.iso -r ~/livecd/
ls -lh /backup/system_image.iso
file /backup/system_image.iso

# 2.3. Точки восстановления системы (аналог через Timeshift)
# ----------------------------------------------------------------
# Обоснование: позволяет откатить систему к предыдущему состоянию
sudo apt install -y timeshift
sudo timeshift --create --comments "Точка восстановления перед экзаменом"
sudo timeshift --list

# 2.4. Создание групп пользователей
# ----------------------------------------------------------------
# Обоснование: разграничение прав доступа разных категорий сотрудников
sudo groupadd developers
sudo groupadd admins
sudo groupadd interns

sudo useradd -m -G developers -s /bin/bash dev_user
sudo useradd -m -G admins -s /bin/bash admin_user
sudo useradd -m -G interns -s /bin/bash intern_user

echo "dev_user:123" | sudo chpasswd
echo "admin_user:123" | sudo chpasswd
echo "intern_user:123" | sudo chpasswd

getent group developers admins interns
id dev_user
id admin_user
id intern_user

# 2.5. Настройка прав доступа к ресурсам
# ----------------------------------------------------------------
# Обоснование: каждый пользователь имеет доступ только к своим ресурсам
sudo mkdir -p /opt/developers /opt/admins /opt/interns /opt/shared

sudo chown root:developers /opt/developers
sudo chown root:admins /opt/admins
sudo chown root:interns /opt/interns
sudo chown root:admins /opt/shared

sudo chmod 770 /opt/developers      # полный доступ для группы developers
sudo chmod 770 /opt/admins          # полный доступ для группы admins
sudo chmod 750 /opt/interns         # чтение для interns, запись только владельцу
sudo chmod 755 /opt/shared          # все читают, пишут только admins

ls -ld /opt/developers /opt/admins /opt/interns /opt/shared

# 2.6. Настройка аутентификации и авторизации
# ----------------------------------------------------------------
# Обоснование: настройка прав на выполнение команд через sudo
echo "%admins ALL=(ALL) ALL" | sudo tee /etc/sudoers.d/admins
echo "%developers ALL=(ALL) /bin/systemctl, /bin/apt" | sudo tee /etc/sudoers.d/developers
echo "interns ALL=(ALL) /bin/ls, /bin/cat" | sudo tee /etc/sudoers.d/interns
sudo visudo -c

sudo -l -U admin_user
sudo -l -U dev_user
sudo -l -U intern_user

# 2.7. Настройка журналов мониторинга
# ----------------------------------------------------------------
# Обоснование: логирование всех действий пользователей для безопасности
sudo apt install -y auditd
sudo systemctl enable auditd
sudo systemctl start auditd

sudo auditctl -w /etc/passwd -p wa -k passwd_changes
sudo auditctl -w /etc/shadow -p wa -k shadow_changes
sudo auditctl -w /opt/developers -p rwxa -k developers_access
sudo auditctl -w /opt/admins -p rwxa -k admins_access

sudo auditctl -l
sudo tail -20 /var/log/auth.log
sudo journalctl -n 20

# ================================================================
# ЧАСТЬ 3: УСТАНОВКА ПО, НАСТРОЙКА И КОНФИГУРИРОВАНИЕ
# ================================================================

# 3.1. Установка Android Studio (IDE для Android)
# ----------------------------------------------------------------
# Обоснование: официальная IDE от Google для разработки под Android
# Поддерживает все типы Android-устройств и языки Kotlin/Java/C++
sudo apt install -y openjdk-17-jdk wget unzip
cd /tmp
wget https://redirector.gvt1.com/edgedl/android/studio/ide-zips/2024.1.1.13/android-studio-2024.1.1.13-linux.tar.gz
sudo tar -xzf android-studio-*.tar.gz -C /opt/
sudo ln -sf /opt/android-studio/bin/studio.sh /usr/local/bin/android-studio
/opt/android-studio/bin/studio.sh &
ls -la ~/Android/Sdk/

# 3.2. Установка Code::Blocks (среда с компиляторами)
# ----------------------------------------------------------------
# Обоснование: легковесная среда с GCC/G++ (измененный набор компиляторов)
# Поддерживает C, C++, Fortran - разные языки программирования
sudo apt install -y codeblocks codeblocks-contrib g++ gcc gfortran
codeblocks &

# 3.3. Установка ClamAV (антивирус - безопасность от хакерских атак)
# ----------------------------------------------------------------
# Обоснование: бесплатный антивирус с открытым кодом, не замедляет систему
sudo apt install -y clamav clamav-daemon clamav-freshclam
sudo systemctl stop clamav-freshclam
sudo freshclam
sudo systemctl start clamav-freshclam
sudo systemctl enable clamav-daemon
sudo systemctl start clamav-daemon
sudo systemctl status clamav-daemon
clamscan --version

# 3.4. Стартовая настройка интерфейса программ (GUI)
# ----------------------------------------------------------------
# Для Android Studio в GUI:
# File → Settings → Appearance & Behavior → Appearance → Theme: Darcula
# File → Settings → Editor → Font → Size: 14
#
# Для Code::Blocks в GUI:
# Settings → Editor → Syntax highlighting → выбрать тему (Dark)
# Settings → Editor → Font → размер 12-14

# 3.5. Настройка обмена данными с другими системами
# ----------------------------------------------------------------
# Git - система контроля версий для обмена кодом с командой
sudo apt install -y git
git config --global user.name "Exam User"
git config --global user.email "exam@example.com"
ssh-keygen -t rsa -b 4096 -C "exam@example.com" -f ~/.ssh/id_rsa -N ""
cat ~/.ssh/id_rsa.pub

# ADB (Android Debug Bridge) - для обмена данными с Android-устройствами
sudo apt install -y adb
echo 'SUBSYSTEM=="usb", ATTR{idVendor}=="18d1", MODE="0666", GROUP="plugdev"' | sudo tee /etc/udev/rules.d/51-android.rules
sudo udevadm control --reload-rules
sudo udevadm trigger
adb kill-server
adb start-server

# 3.6. Настройка параметров совместимости ПО
# ----------------------------------------------------------------

# 3.6.1. Приложения с ограниченной цветовой палитрой (16 бит)
export GDK_COLOR_DEPTH=16

# 3.6.2. Приложения с низким разрешением (800x600, 1024x768)
xrandr --output eDP-1 --mode 1024x768

# 3.6.3. Проблемы с отображением меню и кнопок (программный рендеринг)
export GTK_RENDERER=software
export QT_QUICK_BACKEND=software
export GTK2_RC_FILES=/usr/share/themes/Adwaita/gtk-2.0/gtkrc
export QT_QPA_PLATFORM=xcb

# 3.6.4. Отключение композиции рабочего стола (уменьшение задержки)
gsettings set org.gnome.mutter check-alive-timeout 0
gsettings set org.gnome.desktop.interface enable-animations false

# 3.6.5. Отключение масштабирования при высоком разрешении (HiDPI)
gsettings set org.gnome.desktop.interface scaling-factor 1

# 3.7. Создание скрипта для всех настроек совместимости
# ----------------------------------------------------------------
cat > ~/setup_compatibility.sh << 'EOF'
#!/bin/bash
# Скрипт для применения всех настроек совместимости
gsettings set org.gnome.mutter check-alive-timeout 0 2>/dev/null || true
gsettings set org.gnome.desktop.interface enable-animations false 2>/dev/null || true
gsettings set org.gnome.desktop.interface scaling-factor 1 2>/dev/null || true
export GDK_COLOR_DEPTH=16
export GTK_RENDERER=software
export QT_QUICK_BACKEND=software
export GTK2_RC_FILES=/usr/share/themes/Adwaita/gtk-2.0/gtkrc
export QT_QPA_PLATFORM=xcb
echo "Настройки совместимости применены"
EOF
chmod +x ~/setup_compatibility.sh
./setup_compatibility.sh

# ================================================================
# ЧАСТЬ 4: ДОКУМЕНТИРОВАНИЕ ПО - РУКОВОДСТВО ПОЛЬЗОВАТЕЛЯ
# ================================================================

# Документация пользователя по ГОСТ Р 59795-2021 для Android Studio
# ----------------------------------------------------------------
cat > ~/Руководство_пользователя_Android_Studio.txt << 'EOF'
РУКОВОДСТВО ПОЛЬЗОВАТЕЛЯ
Программное обеспечение: Android Studio
Версия: 2024.1.1
Дата: 25.06.2026

1. НАИМЕНОВАНИЕ И КРАТКОЕ ОПИСАНИЕ
Android Studio — официальная интегрированная среда разработки (IDE) для создания приложений под операционную систему Android. Поддерживает языки Kotlin, Java, C++. Включает редактор кода, эмулятор устройств, отладчик и профилировщик.

2. ЗАПУСК ПРОГРАММЫ
- Через меню приложений: «Activities» → «Android Studio»
- Через терминал: /opt/android-studio/bin/studio.sh
- Или команда: android-studio

3. ЗАПЛАНИРОВАННЫЙ ПРОГРАММОЙ ВЫХОД
- Menu → File → Exit (или Ctrl+Q)
- При незакрытых проектах программа предложит сохранить изменения
- Аварийное завершение: автоматическое сохранение в .idea/saved/

4. ПЕРЕЧЕНЬ ВЫПОЛНЯЕМЫХ ФУНКЦИЙ
- Редактирование кода (подсветка синтаксиса, автодополнение, рефакторинг)
- Сборка (Build) — компиляция кода в APK/AAB
- Эмуляция — запуск виртуального устройства Android
- Отладка — пошаговое выполнение, точки останова, просмотр переменных
- Профилирование — анализ CPU, памяти, сети, энергопотребления
- Управление SDK — установка/обновление платформ и инструментов

5. ОСНОВНЫЕ ПРИЕМЫ РАБОТЫ

5.1. Создание нового проекта
File → New → New Project → Выбрать шаблон (Empty Views Activity) → Указать имя, пакет, минимальную версию SDK → Finish

5.2. Запуск приложения на эмуляторе
Tools → Device Manager → Create device → Выбрать модель телефона → Выбрать образ системы → Нажать «▶» (Run)

5.3. Отладка через точки останова
Кликнуть по номеру строки (красный кружок) → Нажать Debug (жук) → Использовать клавиши: F8 (шаг с обходом), F7 (шаг с заходом), F9 (продолжить)

5.4. Работа с Git
VCS → Enable Version Control Integration → выбрать Git → Commit (Ctrl+K), Push (Ctrl+Shift+K)

6. РЕШЕНИЕ ТИПОВЫХ ПРОБЛЕМ
- Эмулятор не запускается: проверить включенную виртуализацию в BIOS: egrep -c '(vmx|svm)' /proc/cpuinfo
- Gradle sync failed: File → Invalidate Caches / Restart
- Не видит реальное устройство: включить отладку по USB на телефоне, выполнить: adb kill-server && adb start-server
EOF
